NLP Sentiment Analysis 
def predict_sentiment(text):
    positive = ['great','love','excellent']
    negative = ['bad','terrible','hate']
    words = text.lower().split()
    score = sum(1 for w in words if w in positive)
    score -= sum(1 for w in words if w in negative)
    return 'positive' if score > 0 else 'negative'

print(predict_sentiment("I love this product!"))
ML Decision Tree Visualization 
def gini_impurity(labels):
    if len(labels) == 0: return 0
    p = sum(labels) / len(labels)
    return 2 * p * (1 - p)

def info_gain(left, right, parent):
    n = len(left) + len(right)
    return gini_impurity(parent) - (
        len(left)/n * gini_impurity(left) +
        len(right)/n * gini_impurity(right)
    )

# Example: churn prediction
parent = [1,1,0,0,1,0,1,0]  # 1=churn
left  = [1,1,1,1]           # high risk
right = [0,0,0,0]           # low risk
print(f"Info Gain: {info_gain(left, right, parent):.3f}")
DL Neural Network Forward Pass 
import numpy as np

def relu(x): return np.maximum(0, x)

def softmax(x):
    e = np.exp(x - np.max(x))
    return e / e.sum()

# Network: 2 inputs → 3 hidden → 2 outputs
W1 = np.array([[0.2, -0.5], [0.3, 0.8], [-0.1, 0.4]])
b1 = np.array([0.1, -0.2, 0.05])
W2 = np.array([[0.5, -0.3, 0.2], [-0.4, 0.6, -0.1]])
b2 = np.array([0.05, -0.05])

x = np.array([1.0, 0.5])
h = relu(W1 @ x + b1)
out = softmax(W2 @ h + b2)
print(f"Output probabilities: {out}")
NLP Text Tokenization Demo 
text = "AI transforms business operations!"

# Word tokenization
words = text.lower().replace("!","").split()
print(f"Words: {words}")

# Character tokenization
chars = list(text)
print(f"Chars: {chars}")

# Subword (naive BPE-style)
vocab = {"ai", "trans", "form", "s", "busin", "ess", "oper", "ation", "!"}
tokens = []
i = 0
while i < len(text):
    for j in range(len(text), i, -1):
        sub = text[i:j].lower()
        if sub in vocab:
            tokens.append(sub)
            i = j
            break
    else:
        tokens.append(text[i])
        i += 1
print(f"Subwords: {tokens}")
Business Business ROI Calculator 
def calculate_ai_roi(cost, revenue_lift,
                     efficiency_savings, months):
    total_benefit = revenue_lift + efficiency_savings
    roi = ((total_benefit - cost) / cost) * 100
    payback = cost / (total_benefit / months)
    return roi, payback

# Example: Customer churn model
cost = 150_000          # dev + deployment
revenue_lift = 420_000  # retained customers
efficiency = 80_000     # automated workflows
months = 12

roi, payback = calculate_ai_roi(cost, revenue_lift,
                                 efficiency, months)
print(f"ROI: {roi:.1f}%")
print(f"Payback: {payback:.1f} months")
